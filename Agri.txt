    <script>
        // 1. Navigation Logic
        function nav(sectionId) {
            // Hide all sections
            ['home', 'market', 'farmer'].forEach(id => {
                document.getElementById(id).classList.add('hidden');
                document.getElementById(id).classList.remove('block');
            });
            // Show selected
            document.getElementById(sectionId).classList.remove('hidden');
            document.getElementById(sectionId).classList.add('block');
            
            // Update Tab Styles
            document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
            event.target.classList.add('active');
        }

        // 2. Modal Logic
        function toggleModal(modalId) {
            const el = document.getElementById(modalId);
            el.classList.toggle('hidden');
            el.classList.toggle('flex');
        }

        // 3. Simple Login Simulation
        function loginUser() {
            toggleModal('loginModal');
            nav('farmer'); // Redirect to farmer dashboard
            alert("Welcome back! Loading your farm data...");
        }

        // 4. Chart.js Implementation (The Prediction Graph)
        const ctx = document.getElementById('predictionChart').getContext('2d');
        const predictionChart = new Chart(ctx, {
